let myImg = document.querySelectorAll('.project-wrapper .project-img img');
for (let a = 0; a < myImg.length; a++) {
    let loopImg = document.getElementsByClassName("img")[a];
    console.log(loopImg);

    let realWidth = loopImg.naturalWidth;
    let realHeight = loopImg.naturalHeight;


    let original_ratio = realWidth / realHeight
    let new_width = 230 * original_ratio;
    let final_width = Math.round(new_width);

    let new_height = 216 * original_ratio
    let final_height = Math.round(new_height);




    loopImg.style.width = `${final_width}px`;
    loopImg.style.height = `${final_height}px`
}
